The Agent Neural Network view is a real-time, physics-based visualization of an AI swarm. It features a central Orchestrator node connected to specialized sub-agents (Code, UI, Security, Knowledge, etc.). The layout uses a force-directed simulation (repulsion and spring attraction) to create a dynamic, interactive web of connections.

Here is a token-efficient, compact version of the AgentsGraph.tsx component:
code Tsx

import React, { useEffect, useRef, useState } from 'react';
import { Brain, Code2, ShieldCheck, Database, LayoutTemplate, Bug, Globe, Cpu } from 'lucide-react';

const NODES = [
  { id: 'orch', label: 'Orchestrator', icon: Brain, color: '#a855f7', type: 'core' },
  { id: 'code', label: 'Code Expert', icon: Code2, color: '#3b82f6' },
  { id: 'ui', label: 'UI Designer', icon: LayoutTemplate, color: '#ec4899' },
  { id: 'sec', label: 'Security', icon: ShieldCheck, color: '#10b981' },
  { id: 'db', label: 'Knowledge', icon: Database, color: '#f59e0b' },
  { id: 'qa', label: 'QA Tester', icon: Bug, color: '#ef4444' },
  { id: 'web', label: 'Web Search', icon: Globe, color: '#06b6d4' },
  { id: 'sys', label: 'System', icon: Cpu, color: '#6366f1' },
].map(n => ({ ...n, x: Math.random() * 200 - 100, y: Math.random() * 200 - 100, vx: 0, vy: 0 }));

const LINKS = ['code', 'ui', 'sec', 'db', 'qa', 'web', 'sys'].map(id => ({ source: 'orch', target: id }));

export default function AgentsGraph() {
  const container = useRef<HTMLDivElement>(null);
  const [nodes, setNodes] = useState(NODES);
  const [dim, setDim] = useState({ w: 0, h: 0 });

  useEffect(() => {
    const update = () => {
      if (container.current) setDim({ w: container.current.clientWidth, h: container.current.clientHeight });
    };
    update();
    window.addEventListener('resize', update);
    return () => window.removeEventListener('resize', update);
  }, []);

  useEffect(() => {
    let frame = 0;
    const loop = () => {
      setNodes(prev => prev.map((n, i) => {
        let fx = -n.x * 0.01, fy = -n.y * 0.01;
        prev.forEach((o, j) => {
          if (i === j) return;
          const dx = n.x - o.x, dy = n.y - o.y, d = Math.sqrt(dx * dx + dy * dy) || 1;
          const f = 2000 / (d * d);
          fx += (dx / d) * f; fy += (dy / d) * f;
        });
        LINKS.filter(l => l.source === n.id || l.target === n.id).forEach(l => {
          const o = prev.find(p => p.id === (l.source === n.id ? l.target : l.source))!;
          const dx = n.x - o.x, dy = n.y - o.y, d = Math.sqrt(dx * dx + dy * dy) || 1;
          const f = (d - 120) * 0.02;
          fx -= (dx / d) * f; fy -= (dy / d) * f;
        });
        const vx = (n.vx + fx) * 0.9, vy = (n.vy + fy) * 0.9;
        return { ...n, x: n.x + vx, y: n.y + vy, vx, vy };
      }));
      frame = requestAnimationFrame(loop);
    };
    frame = requestAnimationFrame(loop);
    return () => cancelAnimationFrame(frame);
  }, []);

  const cx = dim.w / 2, cy = dim.h / 2;
  return (
    <div ref={container} className="w-full h-full bg-background relative overflow-hidden">
      <svg className="absolute inset-0 w-full h-full opacity-20">
        {LINKS.map((l, i) => {
          const s = nodes.find(n => n.id === l.source)!, t = nodes.find(n => n.id === l.target)!;
          return <line key={i} x1={cx + s.x} y1={cy + s.y} x2={cx + t.x} y2={cy + t.y} stroke="white" strokeWidth="1" />;
        })}
      </svg>
      {nodes.map(n => (
        <div key={n.id} className="absolute -translate-x-1/2 -translate-y-1/2 flex items-center justify-center rounded-full border bg-zinc-900 transition-transform"
          style={{ left: cx + n.x, top: cy + n.y, width: n.type === 'core' ? 56 : 40, height: n.type === 'core' ? 56 : 40, borderColor: n.color }}>
          <n.icon size={n.type === 'core' ? 24 : 18} color={n.color} />
        </div>
      ))}
    </div>
  );
}
